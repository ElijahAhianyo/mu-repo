```python
@contextmanager
def push_dir(directory):
    old = os.path.realpath(os.path.abspath(os.getcwd()))
    new_dir = os.path.realpath(os.path.abspath(directory))
    assert os.path.exists(new_dir)
    os.chdir(new_dir)
    try:
        yield
    finally:
        os.chdir(old)
```
